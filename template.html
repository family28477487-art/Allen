<script setup>

import { ref, computed } from 'vue';

 

// --- 設計與配色 ---

const appBgColor = ref('#F4F7FB'); // 頁面底色延伸（類似冰雪/淺藍色）

const mainContentBgColor = ref('#FFFFFF');

 

// --- 導航與狀態管理 ---

const activeTab = ref('schedule');

const tabs = [

  { id: 'schedule', iconClass: 'fa-solid fa-map-location-dot', name: '行程' },

  { id: 'info', iconClass: 'fa-solid fa-info', name: '資訊' },

  { id: 'shopping', iconClass: 'fa-solid fa-basket-shopping', name: '購物' },

  { id: 'expenses', iconClass: 'fa-solid fa-money-bill-wave', name: '花費' },

];

 

const activeDay = ref('2025-12-01'); // 預設第一天

const days = ref([

    { date: '2025-12-01', week: 'Mon' },

    { date: '2025-12-02', week: 'Tue' },

    { date: '2025-12-03', week: 'Wed' },

    // 更多日期...

]);

 

// --- 行程資訊 (虛擬數據) ---

const categories = ['景點', '住宿', '美食', '購物', '航班', '其他'];

const filterCategory = ref('all');

 

const scheduleData = ref([

    // 第一天行程

    { id: 1, day: '2025-12-01', category: '航班', name: '桃園國際機場 (TPE) 出發', hours: null, price: null, notes: '華航 CI130', location: '桃園國際機場', transport: 'car', travelTime: 'N/A' },

    { id: 2, day: '2025-12-01', category: '租車', name: '新千歲機場租車', hours: '08:00-20:00', price: null, notes: 'Toyota Rent a Car', location: '新千歲機場', transport: 'walk', travelTime: '15分鐘' },

    { id: 3, day: '2025-12-01', category: '住宿', name: '札幌美居飯店', hours: 'Check-in: 15:00', price: null, notes: '位於薄野區', location: '札幌美居飯店', transport: 'car', travelTime: '1小時' },

    // 第二天行程

    { id: 4, day: '2025-12-02', category: '景點', name: '白色戀人公園', hours: '10:00-17:00', price: '800', notes: '網美拍照景點', location: '白色戀人公園', transport: 'car', travelTime: '30分鐘' },

    // 更多行程...

]);

 

const filteredSchedule = computed(() => {

    return scheduleData.value

        .filter(item => item.day === activeDay.value)

        .filter(item => filterCategory.value === 'all' || item.category === filterCategory.value);

});

 

// --- 資訊分頁資料 (虛擬數據) ---

const weather = ref({ temp: 3, feelsLike: -1, icon: 'fa-solid fa-snowflake' });

const weatherIcon = computed(() => weather.value.icon || 'fa-solid fa-cloud');

 

const currentRate = ref(0.22); // 虛擬匯率

const jpyAmount = ref(10000); // 輸入日幣

 

const convertedTWD = computed(() => {

    // 實際應呼叫 API 抓取即時匯率

    if (jpyAmount.value > 0) {

        return (jpyAmount.value * currentRate.value).toFixed(2);

    }

    return 0;

});

 

const flights = ref([

    { id: 1, type: '去程', flightNo: 'CI130', depAirport: 'TPE', depTime: '08:30', arrAirport: 'CTS', arrTime: '13:15', duration: '3h 45m' },

    { id: 2, type: '回程', flightNo: 'CI131', depAirport: 'CTS', depTime: '14:15', arrAirport: 'TPE', arrTime: '18:00', duration: '3h 45m' },

]);

 

const accommodation = ref({

    name: '札幌美居飯店',

    phone: '+81-11-513-1100',

    address: '〒064-0805 北海道札幌市中央区南５条西２丁目４',

});

 

const carRental = ref({

    company: 'Toyota Rent a Car',

    pickupLocation: '新千歲機場營業所',

});

 

 

// --- 購物與花費 (虛擬數據) ---

const shoppingList = ref([

    { id: 101, name: '白色戀人 (盒)', image: null, bought: false },

    { id: 102, name: '薯條三兄弟 (10包)', image: null, bought: true },

]);

 

const expenses = ref([

    { id: 201, category: '住宿', name: '札幌美居飯店 兩晚', date: '2025-12-01', amount: 35000, payment: '信用卡', notes: '已付清' },

    { id: 202, category: '餐飲', name: '湯咖哩', date: '2025-12-02', amount: 3500, payment: '現金', notes: '午餐' },

]);

 

const totalExpense = computed(() => {

    return expenses.value.reduce((sum, item) => sum + item.amount, 0);

});

 

 

// --- 功能函數 ---

 

// 交通 Icon 轉換

const getTrafficIcon = (transport) => {

    switch (transport) {

        case 'walk': return 'fa-solid fa-person-walking';

        case 'car': return 'fa-solid fa-car';

        case 'public': return 'fa-solid fa-bus';

        default: return 'fa-solid fa-route';

    }

};

 

// 獲取前一個行程名稱 (用於計算交通時間)

const getPreviousLocation = (currentIndex) => {

    if (currentIndex === 0) {

        // 每日第一個行程：以前一日住的飯店計算

        return '前一晚住宿地點';

    }

    // 實際開發需根據行程列表獲取前一個行程的名稱

    return '上一個行程';

};

 

// Google Map 導航 (需使用者授權定位)

const openNavigation = (destination) => {

    // 實作：開啟 Google Maps App 或網頁版導航

    const isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);

    if (isMobile) {

        // 嘗試開啟 App

        window.open(`https://maps.apple.com/?daddr=${destination}`, '_blank'); // iOS

    } else {

        // 網頁版

        window.open(`https://www.google.com/maps/dir/?api=1&destination=${destination}`, '_blank');

    }

};

 

 

// --- 模態視窗操作 ---

const isModalOpen = ref(false);

const modalTitle = ref('');

const currentModalType = ref('');

const isEditing = ref(false);

 

const openEditModal = (item) => {

    currentModalType.value = 'schedule';

    isEditing.value = !!item; // 判斷是否為編輯模式

    modalTitle.value = item ? '編輯行程' : '新增行程';

    isModalOpen.value = true;

};

 

const openShoppingModal = () => {

    currentModalType.value = 'shopping';

    modalTitle.value = '新增購物清單項目';

    isModalOpen.value = true;

};

 

const openExpenseModal = () => {

    currentModalType.value = 'expenses';

    modalTitle.value = '新增花費記錄';

    isModalOpen.value = true;

};

 

const confirmDelete = () => {

    if (window.confirm('確定要刪除此行程嗎？此操作不可逆。')) {

        // 執行刪除邏輯

        console.log('行程已刪除');

        isModalOpen.value = false;

    }

};

 

// --- API 整合骨架 ---
